sequenceDiagram
    participant User as User
    participant UI as React UI
    participant Auth as AuthContext
    participant System as System Logic
    participant DB as Firebase Firestore
    
    alt Authentication Flow
        User->>UI: 1. Login / Register
        UI->>Auth: Submit credentials
        Auth->>DB: Query users collection
        DB-->>Auth: Validate credentials
        Auth->>Auth: Session management (30 min timeout)
        Auth->>Auth: Store to localStorage
        Auth-->>UI: Auth token + user data
        UI-->>User: Redirect to Dashboard
    end
    
    alt Donor Workflow
        User->>UI: 2. Access Donor Dashboard
        UI->>System: Check eligibility status
        System->>DB: Query user + donation_history
        DB-->>System: Last donation date
        System->>System: Calculate days since donation
        System-->>UI: Eligibility status & next eligible date
        
        alt If Eligible
            User->>UI: 3. Book Appointment
            UI->>DB: Fetch available slots
            DB-->>UI: Display calendar
            User->>UI: Select date & time
            UI->>DB: Create appointment document
            DB-->>UI: Confirmation (APT ID)
            
            User->>UI: 4. View donation history
            UI->>DB: Query donation_history (donorId)
            DB-->>UI: Timeline of donations
        else If Ineligible
            UI-->>User: Show countdown to next eligible date
        end
    end
    
    alt Staff Workflow
        User->>UI: 5. Access Staff Dashboard
        UI->>DB: Fetch all appointments
        DB-->>UI: Appointments list (filtered by date)
        
        User->>UI: 6. Manage Appointments
        UI->>DB: Update appointment status (pending → confirmed → checked-in → completed)
        DB-->>UI: Status updated
        
        User->>UI: 7. Search Donor
        UI->>DB: Query users (role=donor, filters)
        DB-->>UI: Donor list with contact & blood type
        
        User->>UI: 8. Send donation request
        UI->>DB: Create donation_requests document
        DB-->>UI: Request sent notification
        
        User->>UI: 9. View Blood Inventory
        UI->>DB: Query blood_inventory
        DB-->>UI: All blood types + units + expiration dates
        
        User->>UI: 10. Complete donation
        UI->>DB: Create donation_history record
        UI->>DB: Increment blood_inventory for blood type
        DB-->>UI: Donation recorded
    end
    
    alt Admin Workflow
        User->>UI: 11. Access Admin Dashboard
        UI->>DB: Query appointments + blood_inventory + donation_history
        DB-->>UI: Analytics data & statistics
        
        User->>UI: 12. View Admin Calendar
        UI->>DB: Fetch all appointments
        DB-->>UI: Calendar view of all donations
    end
